## patch for the residue on the lower side of a cutpoint, for loop modeling

NAME coarse_rna_cutpoint_lower
TYPES CUTPOINT_LOWER

BEGIN_SELECTOR
PROPERTY RNA COARSE
NOT VARIANT_TYPE UPPER_TERMINUS
END_SELECTOR

BEGIN_CASE


ADD_ATOM OVL1 VIRT VIRT 0.0
ADD_ATOM OVL2 VIRT VIRT 0.0

## ensure that these guys are considered backbone atoms
SET_BACKBONE_HEAVYATOM OVL1
SET_BACKBONE_HEAVYATOM OVL2

ADD_BOND S    OVL1
ADD_BOND OVL1 OVL2

SET_ICOOR OVL1  113.381000   92.800000    3.832517   S     P     CEN  
SET_ICOOR OVL2 -141.460881   90.300000    4.608955  OVL1   S     P  


END_CASE

## patch for the residue on the upper side of a cutpoint, for loop modeling

NAME protein_centroid_with_HA
TYPES CENTROID_WITH_HA
REPLACE_RES_TYPE
BEGIN_SELECTOR
PROPERTY PROTEIN
#NOT VARIANT_TYPE LOWER_TERMINUS_VARIANT
#NOT VARIANT_TYPE CUTPOINT_UPPER
CMDLINE_SELECTOR CENTROID_HA
END_SELECTOR

BEGIN_CASE ################################# glycine --- taken from GLY in full-atom

BEGIN_SELECTOR
AA GLY
END_SELECTOR

ADD_ATOM 1HA  VIRT VIRT 0.0
ADD_ATOM 2HA  VIRT VIRT 0.0
ADD_BOND CA 1HA
ADD_BOND CA 2HA
SET_ICOOR   1HA   121.400000   70.500000    1.090168   CA    N     C
SET_ICOOR   2HA   117.200000   70.500000    1.089353   CA    N    1HA
END_CASE

BEGIN_CASE ################################# AIB -- does nothing, since AIB has no HA.

BEGIN_SELECTOR
NAME3 AIB DBU DHA
END_SELECTOR

END_CASE

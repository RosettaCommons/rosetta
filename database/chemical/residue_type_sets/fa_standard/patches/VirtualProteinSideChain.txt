)## this patch turns all protein atoms ito virtual atoms
## useful if you want the residues in the pose as placeholders...
NAME Virtual_Protein_SideChain
TYPES    VIRTUAL_SIDE_CHAIN

BEGIN_SELECTOR
PROPERTY PROTEIN
NOT PROPERTY PHOSPHONATE 
# only standard AA's defined for now:
AA ALA ARG ASN ASP CYS CYS CYS GLN GLU GLY HIS HIS ILE LEU LYS MET PHE PRO SER THR TRP TYR VAL
CMDLINE_SELECTOR VIRTUAL_SIDE_CHAIN
END_SELECTOR


BEGIN_CASE ###  ALA ###

BEGIN_SELECTOR
AA ALA
END_SELECTOR

END_CASE

BEGIN_CASE ###  ARG ###

BEGIN_SELECTOR
AA ARG
END_SELECTOR
SET_ATOM_TYPE  CG   VIRT
SET_ATOM_TYPE  CD   VIRT
SET_ATOM_TYPE  NE   VIRT
SET_ATOM_TYPE  CZ   VIRT
SET_ATOM_TYPE  NH1  VIRT
SET_ATOM_TYPE  NH2  VIRT
SET_ATOM_TYPE 1HH1  VIRT
SET_ATOM_TYPE 2HH1  VIRT
SET_ATOM_TYPE 1HH2  VIRT
SET_ATOM_TYPE 2HH2  VIRT
SET_ATOM_TYPE  HE   VIRT
SET_ATOM_TYPE 1HG   VIRT
SET_ATOM_TYPE 2HG   VIRT
SET_ATOM_TYPE 1HD   VIRT
SET_ATOM_TYPE 2HD   VIRT

END_CASE

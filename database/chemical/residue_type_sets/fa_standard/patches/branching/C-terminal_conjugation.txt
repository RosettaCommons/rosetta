## C-terminal conjugation patch for proteins
## currently only implemented for glycine conjugated ubiquitin

NAME C-term_conjugated
TYPES UPPER_TERMINUS_VARIANT
#TYPES BRANCH_LOWER_TERMINUS_VARIANT

BEGIN_SELECTOR
PROPERTY PROTEIN
NOT VARIANT_TYPE BRANCH_LOWER_TERMINUS_VARIANT
NOT VARIANT_TYPE UPPER_TERMINUS_VARIANT
NOT VARIANT_TYPE UPPERTERM_TRUNC_VARIANT
NOT VARIANT_TYPE METHYLATED_CTERMINUS_VARIANT
NOT VARIANT_TYPE C_METHYLAMIDATION
NOT VARIANT_TYPE CUTPOINT_LOWER
END_SELECTOR

BEGIN_CASE ## C-Terminal GLY ##################################################

BEGIN_SELECTOR
AA GLY
END_SELECTOR

ADD_CONNECT C ICOOR   -74.5   68.8   1.4    C   CA   N

SET_POLYMER_CONNECT UPPER NONE

ADD_PROPERTY UPPER_TERMINUS
ADD_PROPERTY BRANCH_LOWER_TERMINUS

END_CASE
